\item Find the distance between the following pair of points
(-13,-3) and (-4,15) with axes inclined at 60 degrees.\\
\solution

let \begin{align}
    \vec{A} = \myvec{-13\\-3}, \vec{B} = \myvec{-4\\15}
\end{align}

And the transform vector(which can be formed by locating where i and j lands after transformation) is given by,
\begin{align}
\vec{X} = \begin{pmatrix}
1 & cos{60}\\
1 & sin{60}
\end{pmatrix}
\end{align}
let normalisation of vector $A$ and vector $B$ after transformation =


 \norm{\vec{X.B}-\vec{X.A}}^2= (\vec{X.B}-\vec{X.A})^\top (\vec{X.B}-\vec{X.A})\;
=  \myvec{18.0 \ 15.588} \myvec{18.0\\15.588}\\
\norm{\vec{X.B}-\vec{X.A}}=\sqrt{((18.0)^2 + (15.588)^2)}\\\nonumber
\norm{\vec{X.B}-\vec{X.A}}=28.11\\\nonumber

Distance between A and B when the axes are inclined at 60 degrees is 28.11
\end{enumerate}
